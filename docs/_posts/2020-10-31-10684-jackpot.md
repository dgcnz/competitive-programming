using namespace std;
int N;
int dp[10010];
int v[10010];
int solve(void) {
  dp[0] = max(v[0], 0);
  for (int i = 1; i < N; ++i)
    dp[i] = max(v[i] + dp[i - 1], v[i]);
  return *max_element(dp, dp + N);
}
int main(void) {
  while (cin >> N and N != 0) {
    for (int i = 0; i < N; ++i)
      cin >> v[i];
    int ans = solve();
    if (ans > 0)
      cout << "The maximum winning streak is " << ans << ".\n";
    else
      cout << "Losing streak.\n";
  }
  return 0;
}
